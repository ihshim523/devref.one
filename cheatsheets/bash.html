# Bash Bash shell scripting quick reference.

Variables

name="Alice"
echo "Hello, $name"
echo "Length: ${#name}"
readonly PI=3.14
export PATH="$HOME/bin:$PATH"

Conditionals

if [[ -f "$file" ]]; then
  echo "File exists"
elif [[ -d "$dir" ]]; then
  echo "Directory exists"
else
  echo "Not found"
fi

[[ "$x" -eq 5 ]] && echo "equal"
[[ "$x" != "" ]] || echo "empty"

Loops

for i in {1..5}; do
  echo "$i"
done

while IFS= read -r line; do
  echo "$line"
done < file.txt

Functions

greet() {
  local name="$1"
  echo "Hello, ${name:-World}"
  return 0
}
greet "Alice"

I/O Redirection

OperatorMeaning
>stdout to file (overwrite)
>>stdout to file (append)
2>stderr to file
&>both to file
<file to stdin
<<<string to stdin
pipe stdout to next

Process Substitution

diff <(ls dir1) <(ls dir2)
while read -r line; do ...; done < <(cmd)
